<<main_loop>>
WHILE expr LOOP
    counter := 'start';

    CASE
        WHEN expr THEN
            msg := 'skip this iteration';
            CONTINUE main_loop;
        ELSE
            msg := 'process normally';
    END CASE;

    counter := 'end';
END LOOP main_loop;
